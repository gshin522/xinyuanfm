async function init() { try { const resp = await fetch(csvUrl); const text = await resp.text(); // 简单的行分割 const lines = text.split(/\r?\n/).filter(line => line.trim() !== ""); allActivities = lines.slice(1).map(line => { // 更加稳健的 CSV 字段分割 const cols = line.split(/,(?=(?:(?:[^"]*"){2})*[^"]*$)/).map(c => c.replace(/^"|"$/g, '').trim()); // 【关键改动】扫描第 14, 15, 16 位（对应 O, P, Q 栏），防止偏移 const statusCheck = (cols[14] || "") + (cols[15] || "") + (cols[16] || ""); return { name: cols[0] || "未命名活动", desc: cols[1] || "", org: cols[2] || "", startDate: cols[3] || "", endDate: cols[4] || "", startTime: cols[5] || "", endTime: cols[6] || "", location: cols[7] || "", state: cols[8] || "", fee: cols[9] || "", contact: cols[10] || "", phone: cols[11] || "", status: statusCheck, poster: cols[16] || cols[17] || "" // 同样海报也往后探一格 }; }).filter(item => item.status.includes("通过") || item.status.includes("批准") || item.status.includes("已通过") ); renderList(allActivities); } catch (e) { console.error("加载错误:", e); showAlert("调取失败，请检查网络或表格发布设置。"); } }
