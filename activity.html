function showDetail(item) { const poster = document.getElementById('posterImg'); // 只有当链接包含 http 且看起来像图片或云端链接时才尝试加载 if (item.poster && item.poster.startsWith('http')) { let id = ""; if (item.poster.includes('/d/')) id = item.poster.split('/d/')[1].split('/')[0]; else if (item.poster.includes('id=')) id = item.poster.split('id=')[1].split('&')[0]; poster.src = id ? `https://drive.google.com/thumbnail?id=${id}&sz=w1000` : item.poster; poster.style.display = 'block'; // 如果图片加载失败（比如是 FB 链接），直接隐藏图片框，不显示破图 poster.onerror = function() { this.style.display = 'none'; }; } else { poster.style.display = 'none'; } // ... 后面的代码保持不变
